Then I created a trace script to capture all the runtime values. Here is the script I wrote to trace_email_forward.py:

```python
#!/usr/bin/env python3
import os
import sys
import subprocess

subprocess.run(["openssl", "genrsa", "-traditional", "-out", "/tmp/test_dkim.key", "1024"], 
               capture_output=True)

os.environ["CONFIG"] = os.path.abspath(
    os.path.join(os.path.dirname(__file__), "tests/test.env")
)
os.environ["DB_URI"] = "postgresql://myuser:mypassword@localhost:5432/simplelogin"
os.environ["NOT_SEND_EMAIL"] = "true"
os.environ["DISABLE_RATE_LIMIT"] = "true"
os.environ["DKIM_PRIVATE_KEY_PATH"] = "/tmp/test_dkim.key"

from email.message import EmailMessage
from email.utils import formatdate, make_msgid
from datetime import datetime
from aiosmtpd.smtp import Envelope

from app.db import Session
from app.models import User, Alias, Mailbox, Contact, EmailLog, SLDomain
from app.email import status
from app.config import EMAIL_DOMAIN
from app.email_utils import generate_reply_email
from app.utils import random_string
from app.mail_sender import mail_sender
import email_handler
from server import create_light_app

app = create_light_app()

with app.app_context():
    sl_domain = SLDomain.get_by(domain=EMAIL_DOMAIN)
    if not sl_domain:
        sl_domain = SLDomain.create(domain=EMAIL_DOMAIN, commit=True)

    test_email = f"testuser_{random_string(8)}@mailbox.test"
    user = User.create(email=test_email, password="testpassword123", 
                       name="Test User for Tracing", activated=True, flush=True)
    mailbox = Mailbox.get_by(user_id=user.id, email=test_email)
    alias_email = f"testalias_{random_string(8)}@{EMAIL_DOMAIN}"
    alias = Alias.create(user_id=user.id, email=alias_email, mailbox_id=mailbox.id, flush=True)
    Session.commit()

    sender_email = "sender@external-domain.com"
    sender_name = "John Doe"
    msg = EmailMessage()
    original_message_id = make_msgid(domain="external.com")
    msg["From"] = f"{sender_name} <{sender_email}>"
    msg["To"] = alias_email
    msg["Subject"] = f"Test Email for Tracing - {random_string(6)}"
    msg["Date"] = formatdate()
    msg["Message-ID"] = original_message_id
    msg.set_content("This is a test email for tracing the forwarding process.")

    envelope = Envelope()
    envelope.mail_from = sender_email
    envelope.rcpt_tos = [alias_email]
    envelope.mail_options = []
    envelope.rcpt_options = []

    mail_sender.purge_stored_emails()
    mail_sender.store_emails_instead_of_sending(True)

    result = email_handler.handle(envelope, msg)
    print(f"SMTP Status: {result}")

    sent_emails = mail_sender.get_stored_emails()
    for sent_email in sent_emails:
        print(f"From Header: {sent_email.msg.get('From')}")
        print(f"Message-ID: {sent_email.msg.get('Message-ID')}")

    contact = Contact.get_by(alias_id=alias.id, website_email=sender_email)
    print(f"Contact ID: {contact.id}")
    print(f"Contact Created At: {contact.created_at}")
    print(f"Reply Email: {contact.reply_email}")
    print(f"New From Header: {contact.new_addr()}")

    email_logs = EmailLog.filter_by(alias_id=alias.id).all()
    latest_log = email_logs[0]
    print(f"EmailLog ID: {latest_log.id}")
    print(f"EmailLog Created At: {latest_log.created_at}")
    print(f"Message ID: {latest_log.message_id}")
    print(f"SL Message ID: {latest_log.sl_message_id}")

    nonexistent_alias = f"nonexistent_{random_string(12)}@{EMAIL_DOMAIN}"
    msg2 = EmailMessage()
    msg2["From"] = f"External Sender <{sender_email}>"
    msg2["To"] = nonexistent_alias
    msg2["Message-ID"] = make_msgid(domain="external.com")
    msg2["Date"] = formatdate()
    msg2.set_content("Test email to non-existent alias.")
    envelope2 = Envelope()
    envelope2.mail_from = sender_email
    envelope2.rcpt_tos = [nonexistent_alias]
    envelope2.mail_options = []
    envelope2.rcpt_options = []
    result2 = email_handler.handle(envelope2, msg2)
    print(f"Non-existent alias SMTP Status: {result2}")

    for contact_email in ["newsletter@company.com", "support@service.io", "john.doe@example.org"]:
        reply_email = generate_reply_email(contact_email, alias)
        print(f"{contact_email} → {reply_email}")

    mail_sender.purge_stored_emails()
    mail_sender.store_emails_instead_of_sending(False)

    EmailLog.filter_by(user_id=user.id).delete()
    Contact.filter_by(user_id=user.id).delete()
    Alias.filter_by(user_id=user.id).delete()
    user.default_mailbox_id = None
    Session.commit()
    Mailbox.filter_by(user_id=user.id).delete()
    User.filter_by(id=user.id).delete()
    Session.commit()
```

The exact log message text that appears when an email is successfully forwarded is "250 Message accepted for delivery" which corresponds to status.E200 in the codebase.

The exact log message text that appears when forwarding fails due to a non-existent alias is "550 SL E515 Email not exist" which corresponds to status.E515.

For the From header transformation, the original From was "John Doe <sender@external-domain.com>" and it got transformed to "John Doe - sender at external-domain.com" <sender_at_external-domain_com_bwxgiykcub@sl.local>. The display name includes the original name plus the email with @ replaced by " at ", and the email address becomes the reverse-alias.

For the Message-ID handling, the original Message-ID <176734830321.52692.15294686223078882115@external.com> is preserved in the forwarded email during the forward phase. The SL Message ID in the EmailLog was None because that only gets generated during the reply phase.

For the database records created during a single forward operation, a Contact record was created with ID 6 at timestamp 2026-01-02 10:05:03.294846+00:00, and an EmailLog record was created with ID 6 at timestamp 2026-01-02 10:05:03.342469+00:00.

The reply email format follows the pattern of replacing @ with _at_ and dots with underscores, then appending a random string and the domain.
