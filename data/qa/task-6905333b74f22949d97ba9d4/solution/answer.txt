To make the code runnable exactly as pinned at this commit, I created an isolated runtime workspace under `/tmp`, built a compatible Python (3.10) locally (the pinned dependency set for this commit is not compatible with the system’s Python 3.11), installed OS dependencies (including Redis), and installed the repo’s pinned Python requirements into a Python 3.10 virtual environment.

```bash
apt-get update -y
apt-get install -y --no-install-recommends build-essential python3-dev redis-server curl optipng \
  libffi-dev libssl-dev libxml2-dev libxslt1-dev zlib1g-dev libsqlite3-dev libbz2-dev libreadline-dev

cd /tmp
curl -fsSL -o Python-3.10.14.tgz https://www.python.org/ftp/python/3.10.14/Python-3.10.14.tgz
tar -xzf Python-3.10.14.tgz
cd Python-3.10.14
./configure --prefix=/tmp/python-3.10.14 --enable-optimizations --with-ensurepip=install
make -j"$(nproc)"
make install

mkdir -p /tmp/paperless-ngx-idle-test-py310
/tmp/python-3.10.14/bin/python3.10 -m venv /tmp/paperless-ngx-idle-test-py310/venv
. /tmp/paperless-ngx-idle-test-py310/venv/bin/activate
grep -vE '^(-i\\s|--extra-index-url\\s)' /app/requirements.txt > /tmp/paperless-req-pypi.txt
PIP_CONFIG_FILE=/dev/null pip install -r /tmp/paperless-req-pypi.txt \
  --index-url https://pypi.org/simple --trusted-host pypi.org --trusted-host files.pythonhosted.org \
  --timeout 120 --retries 10 --no-cache-dir
```
