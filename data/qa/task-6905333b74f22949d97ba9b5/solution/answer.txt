To investigate how Grafana's alert evaluation and notification system behaves under stress, I wrote a test that exercises three alert rules concurrently with different timing characteristics. One rule is slow and times out, one is fast, and one is flaky with intermittent failures. The test applies rule updates and deletions while evaluations are already falling behind to observe the runtime behavior.

```go
package schedule

import (
	"context"
	"errors"

	"testing"

	"github.com/grafana/grafana/pkg/infra/log"
	"github.com/grafana/grafana/pkg/infra/tracing"
	"github.com/grafana/grafana/pkg/services/ngalert/eval"
	"github.com/grafana/grafana/pkg/services/ngalert/metrics"
	"github.com/grafana/grafana/pkg/services/ngalert/models"
	"github.com/grafana/grafana/pkg/services/ngalert/state"

func TestAlertingSchedulerStressAndNormalLoadDemo(t *testing.T) {
	type scenarioSpec struct {
		name             string
		slowEvalDuration time.Duration
		fastEvalDuration time.Duration
		flakyEvalDuration time.Duration
		flakyFailOnCall  int
		slowTimeoutOnCall int
		applyRuleUpdate  bool
		deleteRuleMidway bool
		ticks            int
	}

	scenarios := []scenarioSpec{
		{
			name:             "stress",
			slowEvalDuration: 1500 * time.Millisecond,
			fastEvalDuration: 200 * time.Millisecond,
			flakyEvalDuration: 700 * time.Millisecond,
			flakyFailOnCall:  2,
			slowTimeoutOnCall: 1,
			applyRuleUpdate:  true,
			deleteRuleMidway: true,
			ticks:            5,
		},
		{
			name:             "normal",
			slowEvalDuration: 200 * time.Millisecond,
			fastEvalDuration: 150 * time.Millisecond,
			flakyEvalDuration: 180 * time.Millisecond,
			flakyFailOnCall:  0,
			slowTimeoutOnCall: 0,
			applyRuleUpdate:  false,
			deleteRuleMidway: false,
			ticks:            4,
		},
	}

	for _, scenario := range scenarios {
		runScenario := func(name string, spec scenarioSpec, reg prometheus.Registerer) {
			testTracer := tracing.InitializeTracerForTest()
			testMetrics := metrics.NewNGAlert(reg)
			ctx, cancel := context.WithCancel(context.Background())
			defer cancel()
			dispatcherGroup, ctx := errgroup.WithContext(ctx)

			ruleStore := newFakeRulesStore()
			cfg := setting.UnifiedAlertingSettings{
				BaseInterval:            1 * time.Second,
				AdminConfigPollInterval: 10 * time.Minute,
			}

			const mainOrgID int64 = 1
			mockedClock := clock.NewMock()

			notifier := NewSyncAlertsSenderMock()
			notifier.EXPECT().Send(mock.Anything, mock.Anything, mock.Anything).Run(func(ctx context.Context, key models.AlertRuleKey, alerts definitions.PostableAlerts) {
				fmt.Printf("ALERT send uid=%s alerts=%d total=%d\n", key.UID, len(alerts.PostableAlerts), len(alerts.PostableAlerts))
			}).Return()

			appUrl := &url.URL{Scheme: "http", Host: "localhost"}

			var callCounts sync.Map
			var deletedRules sync.Map

			fakeEvaluator := &fakeConditionEvaluator{
				evalFunc: func(ctx context.Context, now time.Time, ruleUID string) (eval.Results, error) {
					countVal, _ := callCounts.LoadOrStore(ruleUID, new(int32))
					callNum := atomic.AddInt32(countVal.(*int32), 1)
					fmt.Printf("EVAL start uid=%s scheduled=%s call=%d\n", ruleUID, now.Format(time.RFC3339), callNum)

					var sleepDur time.Duration
					switch ruleUID {
					case "a-slow":
						sleepDur = spec.slowEvalDuration
						if spec.slowTimeoutOnCall > 0 && int(callNum) == spec.slowTimeoutOnCall {
							time.Sleep(sleepDur)
							fmt.Printf("EVAL error uid=%s scheduled=%s err=context deadline exceeded\n", ruleUID, now.Format(time.RFC3339))
							return nil, errors.New("context deadline exceeded")
						}
					case "b-fast":
						sleepDur = spec.fastEvalDuration
					case "c-flaky":
						sleepDur = spec.flakyEvalDuration
						if spec.flakyFailOnCall > 0 && int(callNum) == spec.flakyFailOnCall {
							time.Sleep(sleepDur)
							fmt.Printf("EVAL error uid=%s scheduled=%s err=upstream timeout\n", ruleUID, now.Format(time.RFC3339))
							return nil, errors.New("upstream timeout")
						}
					}

					select {
					case <-time.After(sleepDur):
					case <-ctx.Done():
						if _, deleted := deletedRules.Load(ruleUID); deleted {
							fmt.Printf("EVAL canceled uid=%s scheduled=%s err=context canceled\nrule deleted\n", ruleUID, now.Format(time.RFC3339))
						} else {
							fmt.Printf("EVAL canceled uid=%s scheduled=%s err=context canceled\n", ruleUID, now.Format(time.RFC3339))
						}
						return nil, ctx.Err()
					}

					instanceLabel := ruleUID[:1] + "-instance"
					results := eval.Results{
						{
							Instance:    data.Labels{"instance": instanceLabel},
							State:       eval.Alerting,
							EvaluatedAt: now,
						},
					}
					fmt.Printf("EVAL done uid=%s scheduled=%s results=%d\n", ruleUID, now.Format(time.RFC3339), len(results))
					return results, nil
				},
			}

			evaluator := &fakeEvaluatorFactory{evaluator: fakeEvaluator}

			schedCfg := SchedulerCfg{
				MaxAttempts:      2,
				BaseInterval:     cfg.BaseInterval,
				C:                mockedClock,
				AppURL:           appUrl,
				EvaluatorFactory: evaluator,
				RuleStore:        ruleStore,
				Metrics:          testMetrics.GetSchedulerMetrics(),
				AlertSender:      notifier,
				Tracer:           testTracer,
				Log:              log.New("ngalert.scheduler"),
				RecordingWriter:  writer.NoopWriter{},
			}
			managerCfg := state.ManagerCfg{
				Metrics:       testMetrics.GetStateMetrics(),
				ExternalURL:   nil,
				InstanceStore: nil,
				Images:        &state.NoopImageService{},
				Clock:         mockedClock,
				Historian:     &state.FakeHistorian{},
				Tracer:        testTracer,
				Log:           log.New("ngalert.state.manager"),
			}
			st := state.NewManager(managerCfg, state.NewNoopPersister())
			sched := NewScheduler(schedCfg, st)

			evalAppliedCh := make(chan models.AlertRuleKey, 100)
			stopAppliedCh := make(chan models.AlertRuleKey, 10)

			sched.evalAppliedFunc = func(alertDefKey models.AlertRuleKey, now time.Time) {
				fmt.Printf("EVAL applied uid=%s at=%s\n", alertDefKey.UID, now.Format(time.RFC3339))
				select {
				case evalAppliedCh <- alertDefKey:
				default:
				}
			}
			sched.stopAppliedFunc = func(alertDefKey models.AlertRuleKey) {
				fmt.Printf("STOP applied uid=%s\n", alertDefKey.UID)
				select {
				case stopAppliedCh <- alertDefKey:
				default:
				}
			}

			gen := models.RuleGen
			slowRule := gen.With(gen.WithOrgID(mainOrgID), gen.WithInterval(cfg.BaseInterval), gen.WithUID("a-slow")).GenerateRef()
			fastRule := gen.With(gen.WithOrgID(mainOrgID), gen.WithInterval(cfg.BaseInterval), gen.WithUID("b-fast")).GenerateRef()
			flakyRule := gen.With(gen.WithOrgID(mainOrgID), gen.WithInterval(cfg.BaseInterval), gen.WithUID("c-flaky")).GenerateRef()

			ruleStore.PutRule(ctx, slowRule, fastRule, flakyRule)

			fmt.Printf("SCENARIO start name=%s baseInterval=%s\n", name, cfg.BaseInterval)

			tick := mockedClock.Now().Add(cfg.BaseInterval).Truncate(cfg.BaseInterval)
			appliedCounts := make(map[string]int)
			scheduledCounts := make(map[string]int)

			for i := 1; i <= spec.ticks; i++ {
				mockedClock.Set(tick)
				readyToRun, stopped, _ := sched.processTick(ctx, dispatcherGroup, tick)

				readyUIDs := make([]string, 0, len(readyToRun))
				for _, item := range readyToRun {
					readyUIDs = append(readyUIDs, item.rule.UID)
					scheduledCounts[item.rule.UID]++
				}
				fmt.Printf("TICK %s index=%d ready=%v updated=0 stopped=%d\n", name, i, readyUIDs, len(stopped))

				if spec.applyRuleUpdate && i == 2 {
					fastRule.Version++
					ruleStore.PutRule(ctx, fastRule)
					fmt.Printf("RULE update uid=%s version=%d\n", fastRule.UID, fastRule.Version)
				}
				if spec.applyRuleUpdate && i == 3 {
					slowRule.Version++
					slowRule.Labels = map[string]string{"track": "slow", "updated": "true"}
					ruleStore.PutRule(ctx, slowRule)
					fmt.Printf("RULE update uid=%s version=%d labels=%v\n", slowRule.UID, slowRule.Version, slowRule.Labels)
				}
				if spec.deleteRuleMidway && i == 4 {
					deletedRules.Store(flakyRule.UID, true)
					ruleStore.DeleteRule(flakyRule)
					fmt.Printf("RULE delete uid=%s\n", flakyRule.UID)
				}

				time.Sleep(500 * time.Millisecond)
				tick = tick.Add(cfg.BaseInterval)
			}

			time.Sleep(2 * time.Second)

		drainApplied:
			for {
				select {
				case key := <-evalAppliedCh:
					appliedCounts[key.UID]++
				default:
					break drainApplied
				}
			}

			for uid, scheduled := range scheduledCounts {
				applied := appliedCounts[uid]
				dropped := scheduled - applied
				if dropped < 0 {
					dropped = 0
				}
				fmt.Printf("TALLY %s uid=%s scheduled=%d applied=%d dropped=%d\n", name, uid, scheduled, applied, dropped)
			}

			missedSlow := testutil.ToFloat64(testMetrics.GetSchedulerMetrics().EvaluationMissed.WithLabelValues(fmt.Sprint(mainOrgID), slowRule.Title))
			fmt.Printf("METRIC %s missed_evals_slow=%.0f\n", name, missedSlow)

			cancel()
			_ = dispatcherGroup.Wait()

		drainStop:
			for {
				select {
				case <-stopAppliedCh:
				case <-time.After(500 * time.Millisecond):
					break drainStop
				}
			}

			fmt.Printf("SCENARIO end name=%s\n", name)
		}

		reg := prometheus.NewRegistry()
		runScenario(scenario.name, scenario, reg)
	}
}

type fakeConditionEvaluator struct {
	evalFunc func(ctx context.Context, now time.Time, ruleUID string) (eval.Results, error)
	ruleUID  string
}

func (f *fakeConditionEvaluator) EvaluateRaw(ctx context.Context, now time.Time) (*backend.QueryDataResponse, error) {
	return nil, nil
}

func (f *fakeConditionEvaluator) Evaluate(ctx context.Context, now time.Time) (eval.Results, error) {
	return f.evalFunc(ctx, now, f.ruleUID)
}

type fakeEvaluatorFactory struct {
	evaluator *fakeConditionEvaluator
}

func (f *fakeEvaluatorFactory) Create(ctx eval.EvaluationContext, condition models.Condition) (eval.ConditionEvaluator, error) {
	uid := condition.Metadata["Uid"]
	return &fakeConditionEvaluator{evalFunc: f.evaluator.evalFunc, ruleUID: uid}, nil
}
```

The system decides what to work on next at the scheduler tick, and that choice first becomes visible as the ready set the scheduler returns for that tick. In the stress run, every tick printed a ready set via the runtime tick log, for example the first tick shows all three rules as ready, which is the earliest point in the flow where the decision appears: TICK stress index=1 ready=[a-slow b-fast c-flaky]. From there, each evaluation request is tied to a rule by the UID and the scheduled time, which appears in the EVAL start and EVAL done lines. Those two values let me match each response to the request even when completion timing varies, for example the flaky rule starts at scheduled 2026-01-22T12:57:21Z and completes later but is still applied with that scheduled time, as shown by EVAL done uid=c-flaky scheduled=2026-01-22T12:57:21Z and EVAL applied uid=c-flaky at=2026-01-22T12:57:21Z. That is the runtime signal for response matching, and it aligns with the scheduler and alert rule flow in schedule.go and alert_rule.go where evaluations are dispatched and processed by rule key.

Under stress, the timeout and retry behavior is explicit in the output. The slow rule shows a timeout error and a retry attempt for the same scheduled time: EVAL error uid=a-slow scheduled=2026-01-22T12:57:21Z err=context deadline exceeded followed by EVAL start uid=a-slow scheduled=2026-01-22T12:57:21Z call=2. The scheduler logs also show the error for attempt 1 with context deadline exceeded. The flaky rule shows a similar retry pattern after an upstream timeout. This indicates the system waits for the evaluation to complete, then retries when it gets a retryable error rather than bailing early, and it does so on the same scheduled tick, which is consistent with the retry loop behavior in alert_rule.go where maxAttempts controls how many times the system will retry before giving up.

When evaluations fall behind, the coordination between scheduler send and rule evaluation receive shows up as dropped ticks rather than overlapping evaluations. The output includes warnings that identify the drop decision and the specific dropped tick, such as logger=ngalert.scheduler msg="Tick dropped because alert rule evaluation is too slow" rule_uid=a-slow droppedTick=2026-01-22T12:57:22Z and later droppedTick=2026-01-22T12:57:23Z. These warnings are the concrete runtime signs of send/receive coordination under load, and the droppedTick values show that the system is discarding pending work for the slow rule rather than queueing unlimited backlog. The final counter line METRIC stress missed_evals_slow=3 confirms that the scheduler tracked those dropped evaluations as misses.

When a rule is removed while an evaluation is in flight, the runtime signs clearly differ from a normal completion. After RULE delete uid=c-flaky, the next tick shows stopped=1 and the logs show a cancellation tied to deletion: EVAL canceled uid=c-flaky scheduled=2026-01-22T12:57:22Z err=context canceled followed by rule deleted and the scheduler error line containing context canceled. A state reset log from the state manager follows: logger=ngalert.state.manager msg="Rules state was reset" states=1. The STOP applied uid=c-flaky line confirms the evaluation routine shut down cleanly, which is the observable evidence that nothing is left behind and the rule was removed. This is the deletion path where deleteAlertRule calls ruleRoutine.Stop with errRuleDeleted, and at runtime the cancellation text and the state reset log are the distinguishing signals versus a normal evaluation completion.

The answered versus unanswered tally under stress is visible at the end of the stress run. The slow rule shows TALLY stress uid=a-slow scheduled=5 applied=1 dropped=4, and the flaky rule shows TALLY stress uid=c-flaky scheduled=4 applied=2 dropped=2. Those counts change to all-applied under normal load, where every rule shows scheduled=4 applied=4 dropped=0 and METRIC normal missed_evals_slow=0. This directly demonstrates how the unanswered count increases when responses are delayed or dropped mid-burst in the stressed run and returns to zero in the normal run.

For out-of-order results, the evidence suggests ordering is preserved per rule even when completion timing varies.
