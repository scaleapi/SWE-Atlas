I started by turning the prompt into something I could empirically test at runtime: a MinIO server running locally, an IAM user that is intended to be read only for a specific bucket and prefix, and a harness that generates concurrent legitimate writes while the read only identity probes a few "write adjacent" S3 surfaces. I kept the reproduction self contained by using `go run . server ...` for the server and a one off Go probe using `madmin-go` (to create the user and attach policy) and `minio-go` (to issue S3 API calls).

The complete probe script is shown below. It creates a bucket, installs a canned policy that only allows ListBucket within a prefix and GetObject within that same prefix, creates a `readonly` user with that policy, then starts background writer goroutines as the root user while the read only client lists, HEADs, and attempts several mutations (Put, Delete, Copy, multipart initiation, and metadata changes via tagging). The script also adds jitter to the read only client transport so requests see intermittent latency under load.

```go
// tmp_security_policy_probe.go - complete probe script
package main

import (
	"context"
	"crypto/tls"
	"encoding/json"
	"errors"
	"fmt"
	"math/rand"
	"net/http"
	"os"
	"strings"
	"sync"
	"time"

	"github.com/minio/madmin-go/v3"
	"github.com/minio/minio-go/v7"
	"github.com/minio/minio-go/v7/pkg/credentials"
	"github.com/minio/minio-go/v7/pkg/tags"
)

type traceRoundTripper struct {
	base http.RoundTripper
}

func (t *traceRoundTripper) RoundTrip(r *http.Request) (*http.Response, error) {
	fmt.Printf("TRACE_REQ method=%s path=%s\n", r.Method, r.URL.Path)
	resp, err := t.base.RoundTrip(r)
	if err == nil {
		fmt.Printf("TRACE_RESP status=%d %s\n", resp.StatusCode, resp.Status)
	} else {
		fmt.Printf("TRACE_RESP err=%v\n", err)
	}
	return resp, err
}

type delayRoundTripper struct {
	base  http.RoundTripper
	rng   *rand.Rand
	min   time.Duration
	max   time.Duration
	pct   int // 0-100
	label string
}

func (d *delayRoundTripper) RoundTrip(r *http.Request) (*http.Response, error) {
	if d.pct > 0 && d.rng.Intn(100) < d.pct {
		j := d.min + time.Duration(d.rng.Int63n(int64(d.max-d.min)+1))
		time.Sleep(j)
	}
	return d.base.RoundTrip(r)
}

func mustEnv(k string) string {
	v := os.Getenv(k)
	if v == "" {
		panic("missing env var: " + k)
	}
	return v
}

func main() {
	ctx := context.Background()

	endpoint := mustEnv("MINIO_ENDPOINT") // e.g. 127.0.0.1:9000
	rootAccess := mustEnv("MINIO_ROOT_USER")
	rootSecret := mustEnv("MINIO_ROOT_PASSWORD")

	readonlyUser := "readonly"
	readonlySecret := "readonlysecret"

	bucket := "taskgen-sec"
	prefix := "ro/"

	transport := &http.Transport{
		TLSClientConfig: &tls.Config{InsecureSkipVerify: true},
	}

	admin, err := madmin.New(endpoint, rootAccess, rootSecret, false)
	if err != nil {
		panic(err)
	}
	admin.SetCustomTransport(transport)

	s3Root, err := minio.New(endpoint, &minio.Options{
		Creds:     credentials.NewStaticV4(rootAccess, rootSecret, ""),
		Secure:    false,
		Transport: transport,
	})
	if err != nil {
		panic(err)
	}

	// Create bucket (ignore if exists).
	if err := s3Root.MakeBucket(ctx, bucket, minio.MakeBucketOptions{}); err != nil {
		exists, err2 := s3Root.BucketExists(ctx, bucket)
		if err2 != nil || !exists {
			panic(fmt.Errorf("make bucket: %w", err))
		}
	}

	// Create policy: allow ListBucket on bucket within prefix; allow GetObject within prefix.
	// Explicitly deny Put/Delete/Multipart/Copy via missing allows (default deny).
	policy := map[string]any{
		"Version": "2012-10-17",
		"Statement": []map[string]any{
			{
				"Effect":   "Allow",
				"Action":   []string{"s3:GetBucketLocation"},
				"Resource": []string{"arn:aws:s3:::" + bucket},
			},
			{
				"Effect":   "Allow",
				"Action":   []string{"s3:ListBucket"},
				"Resource": []string{"arn:aws:s3:::" + bucket},
				"Condition": map[string]any{
					"StringLike": map[string]any{
						"s3:prefix": []string{prefix + "*"},
					},
				},
			},
			{
				"Effect":   "Allow",
				"Action":   []string{"s3:GetObject"},
				"Resource": []string{"arn:aws:s3:::" + bucket + "/" + prefix + "*"},
			},
		},
	}
	policyBytes, _ := json.Marshal(policy)
	policyName := "taskgen-readonly-prefix"

	if err := admin.AddCannedPolicy(ctx, policyName, policyBytes); err != nil {
		if !strings.Contains(strings.ToLower(err.Error()), "already") {
			panic(fmt.Errorf("add policy: %w", err))
		}
	}

	// Create or update user.
	if err := admin.AddUser(ctx, readonlyUser, readonlySecret); err != nil {
		if !strings.Contains(strings.ToLower(err.Error()), "already") {
			panic(fmt.Errorf("add user: %w", err))
		}
	}
	if err := admin.SetPolicy(ctx, policyName, readonlyUser, false); err != nil {
		panic(fmt.Errorf("set policy: %w", err))
	}

	// Seed object that readonly should be able to read.
	objKey := prefix + "seed.txt"
	_, err = s3Root.PutObject(ctx, bucket, objKey, strings.NewReader("seed"), 4, minio.PutObjectOptions{
		ContentType:  "text/plain",
		UserMetadata: map[string]string{"Owner": "root"},
	})
	if err != nil {
		panic(err)
	}

	// Readonly client with artificial jitter and tracing in the transport.
	rng := rand.New(rand.NewSource(time.Now().UnixNano()))
	roRT := &delayRoundTripper{
		base: &traceRoundTripper{base: transport},
		rng:  rng,
		min:  5 * time.Millisecond,
		max:  80 * time.Millisecond,
		pct:  40,
	}

	s3RO, err := minio.New(endpoint, &minio.Options{
		Creds:     credentials.NewStaticV4(readonlyUser, readonlySecret, ""),
		Secure:    false,
		Transport: roRT,
	})
	if err != nil {
		panic(err)
	}

	s3ROCore, err := minio.NewCore(endpoint, &minio.Options{
		Creds:     credentials.NewStaticV4(readonlyUser, readonlySecret, ""),
		Secure:    false,
		Transport: roRT,
	})
	if err != nil {
		panic(err)
	}

	fmt.Printf("endpoint=%s bucket=%s prefix=%s readonly=%s\n", endpoint, bucket, prefix, readonlyUser)

	// Background writers (root) to create load.
	var writersWG sync.WaitGroup
	writerCtx, cancelWriters := context.WithCancel(ctx)
	for i := 0; i < 8; i++ {
		writersWG.Add(1)
		go func(i int) {
			defer writersWG.Done()
			r := rand.New(rand.NewSource(time.Now().UnixNano() + int64(i)))
			for {
				select {
				case <-writerCtx.Done():
					return
				default:
				}
				k := fmt.Sprintf("%sload/%d-%d.txt", prefix, i, r.Intn(1_000_000))
				body := strings.NewReader("load")
				_, _ = s3Root.PutObject(ctx, bucket, k, body, 4, minio.PutObjectOptions{ContentType: "text/plain"})
				time.Sleep(time.Duration(r.Intn(30)) * time.Millisecond)
			}
		}(i)
	}

	// 1) List within prefix.
	listed := 0
	for obj := range s3RO.ListObjects(ctx, bucket, minio.ListObjectsOptions{Prefix: prefix, Recursive: true, MaxKeys: 5}) {
		if obj.Err != nil {
			fmt.Printf("LIST err=%v\n", obj.Err)
			break
		}
		fmt.Printf("LIST key=%s size=%d etag=%s lastmod=%s\n", obj.Key, obj.Size, obj.ETag, obj.LastModified.UTC().Format(time.RFC3339Nano))
		listed++
		if listed >= 5 {
			break
		}
	}

	// 2) HEAD/Stat metadata.
	st, err := s3RO.StatObject(ctx, bucket, objKey, minio.StatObjectOptions{})
	if err != nil {
		fmt.Printf("HEAD key=%s err=%v\n", objKey, err)
	} else {
		fmt.Printf("HEAD key=%s size=%d etag=%s contentType=%s usermeta=%v\n", st.Key, st.Size, st.ETag, st.ContentType, st.UserMetadata)
	}

	// 3) Attempt PUT (should be denied).
	_, err = s3RO.PutObject(ctx, bucket, prefix+"ro-put.txt", strings.NewReader("nope"), 4, minio.PutObjectOptions{ContentType: "text/plain"})
	fmt.Printf("PUT as readonly err=%v\n", err)

	// 4) Attempt DELETE (should be denied).
	err = s3RO.RemoveObject(ctx, bucket, objKey, minio.RemoveObjectOptions{})
	fmt.Printf("DELETE as readonly err=%v\n", err)

	// 5) Attempt COPY into prefix (should be denied).
	dst := minio.CopyDestOptions{Bucket: bucket, Object: prefix + "ro-copy.txt"}
	src := minio.CopySrcOptions{Bucket: bucket, Object: objKey}
	_, err = s3RO.CopyObject(ctx, dst, src)
	fmt.Printf("COPY as readonly err=%v\n", err)

	// 6) Attempt multipart upload initiation (should be denied).
	uploadID, err := s3ROCore.NewMultipartUpload(ctx, bucket, prefix+"ro-mpu.bin", minio.PutObjectOptions{ContentType: "application/octet-stream"})
	fmt.Printf("MPU init as readonly uploadID=%q err=%v\n", uploadID, err)

	// 7) Attempt metadata change via PutObjectTagging (should be denied).
	tagMap, _ := tags.NewTags(map[string]string{"tamper": "attempt"}, false)
	err = s3RO.PutObjectTagging(ctx, bucket, objKey, tagMap, minio.PutObjectTaggingOptions{})
	fmt.Printf("PUT_TAGGING as readonly err=%v\n", err)

	// 8) Attempt metadata replacement via CopyObject with ReplaceMetadata (should be denied).
	dstMeta := minio.CopyDestOptions{
		Bucket:          bucket,
		Object:          objKey,
		UserMetadata:    map[string]string{"Modified": "yes"},
		ReplaceMetadata: true,
	}
	srcMeta := minio.CopySrcOptions{Bucket: bucket, Object: objKey}
	_, err = s3RO.CopyObject(ctx, dstMeta, srcMeta)
	fmt.Printf("METADATA_REPLACE as readonly err=%v\n", err)

	// Small pause and stop writers.
	time.Sleep(250 * time.Millisecond)
	cancelWriters()
	writersWG.Wait()

	// Check whether any forbidden objects exist (side effects).
	checkKeys := []string{prefix + "ro-put.txt", prefix + "ro-copy.txt", prefix + "ro-mpu.bin"}
	for _, k := range checkKeys {
		_, err := s3Root.StatObject(ctx, bucket, k, minio.StatObjectOptions{})
		if err == nil {
			fmt.Printf("SIDE_EFFECT key=%s exists=true\n", k)
			continue
		}
		var resp minio.ErrorResponse
		if errors.As(err, &resp) {
			fmt.Printf("SIDE_EFFECT key=%s exists=false status=%d code=%s\n", k, resp.StatusCode, resp.Code)
		} else {
			fmt.Printf("SIDE_EFFECT key=%s exists=unknown err=%v\n", k, err)
		}
	}

	// Check that seed object metadata was NOT modified.
	stAfter, err := s3Root.StatObject(ctx, bucket, objKey, minio.StatObjectOptions{})
	if err != nil {
		fmt.Printf("METADATA_CHECK err=%v\n", err)
	} else {
		fmt.Printf("METADATA_CHECK key=%s usermeta=%v (should still be Owner:root)\n", stAfter.Key, stAfter.UserMetadata)
	}
}
```

To run it, start MinIO with default root credentials on localhost and then execute the probe:

```bash
# Terminal 1: Start MinIO server
MINIO_ROOT_USER=minioadmin MINIO_ROOT_PASSWORD=minioadmin \
  go run . server /tmp/taskgen-minio-data --address 127.0.0.1:9000 --console-address 127.0.0.1:9001

# Terminal 2: Run the probe (after server is ready)
MINIO_ENDPOINT=127.0.0.1:9000 MINIO_ROOT_USER=minioadmin MINIO_ROOT_PASSWORD=minioadmin \
  go run ./tmp_security_policy_probe.go

# Terminal 2: Cleanup after probe completes
rm ./tmp_security_policy_probe.go
```

The first thing I wanted to confirm was what the read only identity can learn from listing and HEAD style operations without performing any mutation. Under concurrent background writers, the read only identity was able to list within the allowed prefix and see object keys, sizes, ETags, and last modified timestamps. The HTTP trace shows the request path and response status:

```
TRACE_REQ method=GET path=/taskgen-sec?list-type=2&max-keys=5&prefix=ro%2F
TRACE_RESP status=200 200 OK
endpoint=127.0.0.1:9000 bucket=taskgen-sec prefix=ro/ readonly=readonly
LIST key=ro/load/0-239828.txt size=4 etag=ec4d1eb36b22d19728e9d1d23ca84d1c lastmod=2026-01-23T22:01:37.142Z
LIST key=ro/load/1-788944.txt size=4 etag=ec4d1eb36b22d19728e9d1d23ca84d1c lastmod=2026-01-23T22:01:37.155Z
LIST key=ro/load/2-452895.txt size=4 etag=ec4d1eb36b22d19728e9d1d23ca84d1c lastmod=2026-01-23T22:01:37.146Z
LIST key=ro/load/3-243946.txt size=4 etag=ec4d1eb36b22d19728e9d1d23ca84d1c lastmod=2026-01-23T22:01:37.141Z
LIST key=ro/load/4-572167.txt size=4 etag=ec4d1eb36b22d19728e9d1d23ca84d1c lastmod=2026-01-23T22:01:37.148Z
```

The read only identity was also able to issue a HEAD equivalent (`StatObject`) on an object inside the prefix and see content type and user metadata in the response. In this reproduction I intentionally stored an `x-amz-meta-owner=root` metadata key on the seed object to see whether it leaks through to the read only principal, and it did:

```
TRACE_REQ method=HEAD path=/taskgen-sec/ro/seed.txt
TRACE_RESP status=200 200 OK
HEAD key=ro/seed.txt size=4 etag=fe4c0f30aa359c41d9f9a5f69c8c4192 contentType=text/plain usermeta=map[Owner:root]
```

Next I tested several "write adjacent" actions from the read only identity while the background writers continued to write objects under the same bucket and prefix. Each attempted mutation was rejected with `Access Denied.` (HTTP 403) in the client error, even though the system was under concurrency and the read only transport introduced intermittent request latency:

```
TRACE_REQ method=PUT path=/taskgen-sec/ro/ro-put.txt
TRACE_RESP status=403 403 Forbidden
PUT as readonly err=Access Denied.
TRACE_REQ method=DELETE path=/taskgen-sec/ro/seed.txt
TRACE_RESP status=403 403 Forbidden
DELETE as readonly err=Access Denied.
TRACE_REQ method=PUT path=/taskgen-sec/ro/ro-copy.txt
TRACE_RESP status=403 403 Forbidden
COPY as readonly err=Access Denied.
TRACE_REQ method=POST path=/taskgen-sec/ro/ro-mpu.bin?uploads=
TRACE_RESP status=403 403 Forbidden
MPU init as readonly uploadID="" err=Access Denied.
```

I also tested metadata modification specifically, which the prompt calls out as a "write adjacent" surface. I attempted to add object tags via PutObjectTagging and to replace user metadata via CopyObject with ReplaceMetadata. Both were denied:

```
TRACE_REQ method=PUT path=/taskgen-sec/ro/seed.txt?tagging=
TRACE_RESP status=403 403 Forbidden
PUT_TAGGING as readonly err=Access Denied.
TRACE_REQ method=PUT path=/taskgen-sec/ro/seed.txt
TRACE_RESP status=403 403 Forbidden
METADATA_REPLACE as readonly err=Access Denied.
```

Finally, to ensure there were no silent side effects from partially accepted requests, I checked for the existence of the would be created objects using the root client after the read only attempts completed. All three mutation targets were absent and returned `NoSuchKey`:

```
SIDE_EFFECT key=ro/ro-put.txt exists=false status=404 code=NoSuchKey
SIDE_EFFECT key=ro/ro-copy.txt exists=false status=404 code=NoSuchKey
SIDE_EFFECT key=ro/ro-mpu.bin exists=false status=404 code=NoSuchKey
```

I also verified that the seed object's metadata remained unchanged after the metadata modification attempts:

```
METADATA_CHECK key=ro/seed.txt usermeta=map[Owner:root] (should still be Owner:root)
```

In this run, the policy boundary held under load: the read only identity could list and learn metadata (including user metadata) for objects it can access, but it could not mutate state via PUT, DELETE, server side COPY, multipart upload initiation, tagging, or metadata replacement. I did not observe any transient acceptance of forbidden operations during concurrent background writes and client side induced latency.
