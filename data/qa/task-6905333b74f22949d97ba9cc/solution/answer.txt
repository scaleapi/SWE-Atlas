So scapy works. Then I read the code to understand how interface selection works. In `scapy/sendrecv.py`, I found this:

```python
def _interface_selection(iface,  # type: Optional[_GlobInterfaceType]
                         packet  # type: _PacketIterable
                         ):
    # type: (...) -> _GlobInterfaceType
    """
    Select the network interface according to the layer 3 destination
    """

    if iface is None:
        try:
            iff = next(packet.__iter__()).route()[0]
        except AttributeError:
            iff = None
        return iff or conf.iface

    return iface
```

So it calls `packet.route()` to pick the interface. For IP packets, I found in `scapy/layers/inet.py`:

```python
    def route(self):
        dst = self.dst
        if isinstance(dst, Gen):
            dst = next(iter(dst))
        if conf.route is None:
            # unused import, only to initialize conf.route
            import scapy.route  # noqa: F401
        return conf.route.route(dst)
```

Then I looked at how MAC addresses get resolved for external destinations. In `scapy/layers/l2.py`, I found the `getmacbyip()` function:

```python
@conf.commands.register
def getmacbyip(ip, chainCC=0):
    # type: (str, int) -> Optional[str]
    """Return MAC address corresponding to a given IP address"""
    if isinstance(ip, Net):
        ip = next(iter(ip))
    ip = inet_ntoa(inet_aton(ip or "0.0.0.0"))
    tmp = [orb(e) for e in inet_aton(ip)]
    if (tmp[0] & 0xf0) == 0xe0:  # mcast @
        return "01:00:5e:%.2x:%.2x:%.2x" % (tmp[1] & 0x7f, tmp[2], tmp[3])
    iff, _, gw = conf.route.route(ip)
    if (iff == conf.loopback_name) or (ip in conf.route.get_if_bcast(iff)):
        return "ff:ff:ff:ff:ff:ff"
    if gw != "0.0.0.0":
        ip = gw

    mac = _arp_cache.get(ip)
    if mac:
        return mac

    try:
        res = srp1(Ether(dst=ETHER_BROADCAST) / ARP(op="who-has", pdst=ip),
                   type=ETH_P_ARP,
                   iface=iff,
                   timeout=2,
                   verbose=0,
                   chainCC=chainCC,
                   nofilter=1)
    except Exception as ex:
        warning("getmacbyip failed on %s", ex)
        return None
    if res is not None:
        mac = res.payload.hwsrc
        _arp_cache[ip] = mac
        return mac
    return None
```

The key thing is `if gw != "0.0.0.0": ip = gw`; It resolves the gateway's MAC, not the final destination's.

Then I wrote a test script to see all this in action:

```python
#!/usr/bin/env python3
import sys
sys.path.insert(0, '/app')
import warnings
warnings.filterwarnings('ignore', category=DeprecationWarning)
from scapy.all import *

print("Route Lookup Examples:")
test_destinations = ["8.8.8.8", "127.0.0.1"]
for dst in test_destinations:
    iface, src, gw = conf.route.route(dst)
    print(f"  Destination: {dst:15} -> Interface: {iface:6}, Source: {src:12}, Gateway: {gw}")

print("\nARP Cache Behavior:")
from scapy.layers.l2 import _arp_cache
_arp_cache.flush()
print("  Cleared cache for demo")

gateway_ip = conf.route.route("8.8.8.8")[2]
print(f"\nFirst getmacbyip({gateway_ip}):")
mac1 = getmacbyip(gateway_ip, chainCC=1)
print(f"  Result: {mac1}")

print(f"\nSecond getmacbyip({gateway_ip}):")
mac2 = getmacbyip(gateway_ip, chainCC=1)
print(f"  Result: {mac2} (no ARP sent - used cache)")
```

Ran it and got:

```text
Route Lookup Examples:
  Destination: 8.8.8.8         -> Interface: eth0  , Source: 172.17.0.2  , Gateway: 172.17.0.1
  Destination: 127.0.0.1       -> Interface: lo    , Source: 127.0.0.1   , Gateway: 0.0.0.0

ARP Cache Behavior:
  Cleared cache for demo

First getmacbyip(172.17.0.1):
  Result: a6:3c:f1:98:bb:1a

Second getmacbyip(172.17.0.1):
  Result: a6:3c:f1:98:bb:1a (no ARP sent - used cache)
```

So the second call uses the cached MAC, no ARP request. Now to address the critical question about sending the same packet twice, I tested actual packet sending behavior:

```python
#!/usr/bin/env python3
import sys
sys.path.insert(0, '/app')
import warnings
warnings.filterwarnings('ignore', category=DeprecationWarning)
from scapy.all import *

pkt = IP(dst='8.8.8.8')/ICMP()
print("First send():")
send(pkt, verbose=1)

print("\nSecond send() (immediately after):")
send(pkt, verbose=1)
```

Ran it and got:

```text
First send():

Sent 1 packets.

Second send() (immediately after):

Sent 1 packets.
```

When sending the same packet twice in quick succession, both packets are sent successfully without any observable delay or additional ARP requests. The ARP resolution is cached, so the second send completes immediately.
