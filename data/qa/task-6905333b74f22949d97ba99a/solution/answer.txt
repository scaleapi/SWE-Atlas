```typescript
import '@automattic/calypso-polyfills';
import { createExPlatClient } from '../src/create-explat-client';
import type { Config } from '../src/types';

(global as any).window = {};

console.log("Test: Server Timeout Behavior");
console.log("=============================\n");

async function runTest(): Promise<void> {
    const errors: string[] = [];
    
    const config: Config = {
        logError: (error) => errors.push(error.message),
        fetchExperimentAssignment: async () => {
            console.log("[Fetch] Server is not responding...");
            return new Promise(() => {}); // Never resolves
        },
        getAnonId: async () => 'user-123',
        isDevelopmentMode: false
    };
    
    const client = createExPlatClient(config);
    
    console.log("Calling loadExperimentAssignment...\n");
    const startTime = Date.now();
    
    let threwException = false;
    let result;
    try {
        result = await client.loadExperimentAssignment('my_experiment');
    } catch (e) {
        threwException = true;
        console.log("EXCEPTION THROWN:", e);
    }
    
    const elapsed = Date.now() - startTime;
    
    console.log("Results:");
    console.log("  Time elapsed: " + elapsed + "ms");
    console.log("  Threw exception: " + threwException);
    
    if (result) {
        console.log("  Response received: yes");
        console.log("  Response object:");
        console.log("    experimentName: " + result.experimentName);
        console.log("    variationName: " + result.variationName);
        console.log("    ttl: " + result.ttl + " seconds");
        console.log("    isFallbackExperimentAssignment: " + result.isFallbackExperimentAssignment);
    }
    
    console.log("  Errors logged internally: " + errors.length);
    if (errors.length > 0) {
        console.log("    Message: " + errors[0]);
    }
}

runTest().catch(console.error);
```

```typescript
import '@automattic/calypso-polyfills';
import { createExPlatClient } from '../src/create-explat-client';
import type { Config } from '../src/types';

(global as any).window = {};

console.log("Test: Concurrent Requests for Same Experiment");
console.log("=============================================\n");

async function runTest(): Promise<void> {
    let networkCallCount = 0;
    
    const config: Config = {
        logError: () => {},
        fetchExperimentAssignment: async ({ experimentName }) => {
            networkCallCount++;
            console.log("[Network Call #" + networkCallCount + "] Fetching " + experimentName);
            await new Promise(resolve => setTimeout(resolve, 200));
            return {
                ttl: 60,
                variations: { [experimentName]: 'treatment' }
            };
        },
        getAnonId: async () => 'user-789',
        isDevelopmentMode: false
    };
    
    const client = createExPlatClient(config);
    
    console.log("Firing 5 concurrent requests for the same experiment...\n");
    
    const promise1 = client.loadExperimentAssignment('button_color_exp');
    const promise2 = client.loadExperimentAssignment('button_color_exp');
    const promise3 = client.loadExperimentAssignment('button_color_exp');
    const promise4 = client.loadExperimentAssignment('button_color_exp');
    const promise5 = client.loadExperimentAssignment('button_color_exp');
    
    const results = await Promise.all([promise1, promise2, promise3, promise4, promise5]);
    
    console.log("\nResults:");
    console.log("  Application made: 5 concurrent requests");
    console.log("  Actual network calls: " + networkCallCount);
    console.log("  All got same variation: " + results.every(r => r.variationName === results[0].variationName));
    console.log("  Variation received: " + results[0].variationName);
}

runTest().catch(console.error);
```

```typescript
import '@automattic/calypso-polyfills';
import { createExPlatClient } from '../src/create-explat-client';
import type { Config } from '../src/types';

(global as any).window = {};

console.log("Test: Synchronous Getter Before Async Load");
console.log("==========================================\n");

async function runTest(): Promise<void> {
    const errors: string[] = [];
    
    const config: Config = {
        logError: (error) => errors.push(error.message),
        fetchExperimentAssignment: async ({ experimentName }) => {

            return {
                ttl: 60,
                variations: { [experimentName]: 'treatment' }
            };
        },
        getAnonId: async () => 'user-sync',
        isDevelopmentMode: true
    };
    
    const client = createExPlatClient(config);
    
    console.log("Case 1: Call sync getter without ever calling async load");
    errors.length = 0;
    
    let threwException = false;
    let syncResult;
    try {
        syncResult = client.dangerouslyGetExperimentAssignment('never_loaded');
    } catch (e) {
        threwException = true;
        console.log("  EXCEPTION: " + e);
    }
    
    console.log("  Threw exception: " + threwException);
    if (syncResult) {
        console.log("  Returned a result: yes");
        console.log("  variationName: " + syncResult.variationName);
        console.log("  isFallbackExperimentAssignment: " + syncResult.isFallbackExperimentAssignment);
    }
    console.log("  Errors logged: " + errors.length);
    if (errors.length > 0) console.log("  Error message: " + errors[0]);
    
    console.log("\nCase 2: Start async load, then immediately call sync getter");
    errors.length = 0;
    
    const loadPromise = client.loadExperimentAssignment('loading_exp');
    console.log("  Async load started (not awaited)");
    
    const syncResult2 = client.dangerouslyGetExperimentAssignment('loading_exp');
    console.log("  Sync getter returned:");
    console.log("    variationName: " + syncResult2.variationName);
    console.log("    isFallbackExperimentAssignment: " + syncResult2.isFallbackExperimentAssignment);
    console.log("  Errors logged: " + errors.length);
    
    const asyncResult = await loadPromise;
    console.log("  After async load completed:");
    console.log("    variationName: " + asyncResult.variationName);
    console.log("    isFallbackExperimentAssignment: " + asyncResult.isFallbackExperimentAssignment);
    
    console.log("\nCase 3: Call sync getter after async load completed");
    errors.length = 0;
    
    const syncResult3 = client.dangerouslyGetExperimentAssignment('loading_exp');
    console.log("  Sync getter returned:");
    console.log("    variationName: " + syncResult3.variationName);
    console.log("    isFallbackExperimentAssignment: " + syncResult3.isFallbackExperimentAssignment);
    console.log("  Matches async result: " + (syncResult3.variationName === asyncResult.variationName));
    
    console.log("\nConclusion:");
    console.log("  Does sync getter break the application? NO");
    console.log("  What does it return before load? Fallback with null variation");
    console.log("  What does it return after load? The actual assignment");
```

Summary of findings:

The client never throws exceptions. When the server is unavailable or times out, the client returns a fallback response with variationName null, ttl 60 seconds, and isFallbackExperimentAssignment true. The user gets assigned to null variation (default experience).

Concurrent requests are deduplicated. Five simultaneous requests for the same experiment result in only one network call.

Caching works with a minimum 60 second TTL. Requests in quick succession use the cached value. After the cache expires (60+ seconds), a new network call is made.

The synchronous getter does not break the application when called before async load. It returns a fallback and logs an error internally. After load completes, it returns the actual assignment.
