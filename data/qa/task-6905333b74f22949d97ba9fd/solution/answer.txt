Then I simulated an actual file transfer using the TestFileTransmission class:

```python
#!/usr/bin/env python3
import sys
import os
sys.path.insert(0, '/app')

from kitty.file_transmission import FileTransmissionCommand, Action, FileType, TransmissionType, TestFileTransmission

# Create test file
TEST_FILE = '/tmp/kitty_transfer_demo/source_file.txt'
with open(TEST_FILE, 'wb') as f:
    for i in range(200):
        f.write(f"Line {i:05d}: This is test content.\n".encode())

# Use TestFileTransmission to simulate terminal side
ft = TestFileTransmission(allow=True)

# Step 1: Send action
send_cmd = FileTransmissionCommand(action=Action.send, id='test123')
ft.handle_serialized_command(memoryview(send_cmd.serialize().encode()))

# Step 2: File metadata
file_cmd = FileTransmissionCommand(
    action=Action.file, id='test123', file_id='file1',
    name='/tmp/dest.txt', size=11400, ftype=FileType.regular
)
ft.handle_serialized_command(memoryview(file_cmd.serialize().encode()))

# Step 3: Send data chunks
with open(TEST_FILE, 'rb') as f:
    content = f.read()

CHUNK_SIZE = 4096
offset = 0
while offset < len(content):
    chunk = content[offset:offset + CHUNK_SIZE]
    is_last = offset + len(chunk) >= len(content)
    data_cmd = FileTransmissionCommand(
        action=Action.end_data if is_last else Action.data,
        id='test123', file_id='file1', data=chunk
    )
    ft.handle_serialized_command(memoryview(data_cmd.serialize().encode()))
    offset += len(chunk)

# Step 4: Finish
ft.handle_serialized_command(memoryview(FileTransmissionCommand(action=Action.finish, id='test123').serialize().encode()))
```

Then I wrote a script to demonstrate delta transfer efficiency:

```python
#!/usr/bin/env python3
import sys
import os
import io
sys.path.insert(0, '/app')

from kittens.transfer.rsync import Patcher, Differ

# Create 50KB original file
with open('/tmp/original.txt', 'wb') as f:
    for i in range(1000):
        f.write(f"Line {i:04d}: This is the original content.\n".encode())

# Create modified version - change 2% (lines 100-120)
with open('/tmp/original.txt', 'rb') as f:
    content = f.readlines()

for i in range(100, 120):
    content[i] = f"Line {i:04d}: THIS LINE HAS BEEN MODIFIED!!!\n".encode()

with open('/tmp/modified.txt', 'wb') as f:
    f.writelines(content)

# Create signature of original
orig_size = os.path.getsize('/tmp/original.txt')
patcher = Patcher(orig_size)
sig_buf = bytearray(32)
sig_data = bytearray()
sig_data.extend(sig_buf[:patcher.signature_header(sig_buf)])

with open('/tmp/original.txt', 'rb') as f:
    block_buf = bytearray(patcher.block_size)
    while True:
        n = f.readinto(block_buf)
        if n == 0:
            break
        sig_data.extend(sig_buf[:patcher.sign_block(memoryview(block_buf)[:n], sig_buf)])

# Compute delta
differ = Differ()
differ.add_signature_data(bytes(sig_data))
differ.finish_signature_data()

delta_buf = io.BytesIO()
with open('/tmp/modified.txt', 'rb') as f:
    mod_data = io.BytesIO(f.read())

while differ.next_op(lambda b: mod_data.readinto(b), lambda d: delta_buf.write(bytes(d))):
    pass

delta = delta_buf.getvalue()
```
